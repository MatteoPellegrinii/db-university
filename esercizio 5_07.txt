ESERCIZIO J0IN

1) SELECT DISTINCT * FROM `students`
JOIN degrees
ON students.degree_id = degrees.id
WHERE degrees.name = "Corso di laurea in economia"

2) SELECT * FROM `courses`
JOIN departments JOIN degrees
ON courses.degree_id = degrees.id
AND degrees.department_id = departments.id
WHERE departments.name = "Dipartimento di neuroscienze"
AND degrees.level = "Magistrale"

3) SELECT * FROM `courses`
JOIN course_teacher
JOIN teachers
ON teachers.id = course_teacher.teacher_id
AND course_teacher.course_id = courses.id
WHERE teachers.id = 44

4) SELECT * FROM `students`
JOIN degrees
JOIN departments
ON students.degree_id = degrees.id
AND degrees.department_id = departments.id
ORDER BY students.surname

5)  SELECT * FROM `degrees`
JOIN course_teacher
JOIN teachers
JOIN courses
ON teachers.id = course_teacher.teacher_id
AND course_teacher.course_id = courses.id
AND courses.degree_id = degrees.id
ORDER BY degrees.name

6) SELECT * FROM teachers
JOIN courses
JOIN course_teacher
JOIN degrees
JOIN departments
ON teachers.id = course_teacher.teacher_id
AND course_teacher.course_id = courses.id
AND courses.degree_id = degrees.id
AND degrees.department_id = departments.id
WHERE departments.name = "Dipartimento di matematica" 	

ESERCIZIO GROUP_BY
1) SELECT YEAR(`enrolment_date`) as anno, COUNT(`id`) as numero_studenti FROM `students` GROUP BY YEAR(`enrolment_date`)

2) SELECT `office_address`, COUNT(`id`) as numero_insegnanti FROM `teachers` GROUP BY `office_address`

3) SELECT `exam_id`, ROUND(AVG(`vote`),0) as media_voti FROM `exam_student` GROUP BY `exam_id`

4) SELECT `department_id`, COUNT(`id`) as numero_corsi_di_laurea FROM `degrees` GROUP BY `department_id`
